function save_result(){
if ($('input[name="selected_value"]:checked').val()){
$.ajax({
type: "POST",
url: $SCRIPT_ROOT + '/save_result',
data: JSON.stringify({
query: $.trim($('input[name="query"]').val()),
selected: $('input[name="selected_value"]:checked').val(),
notes: $.trim($("#comment").val())
}),
contentType: "application/json",
dataType: "text",
success: function(){
toastr.success("Data successfully Saved");
},
failure: function(jqXHR) {
alert(jqXHR.responseText);
}
});
} else {
alert("Please select a value from top matches")
}

}

function display_top_values(row_id){
if($('#top'+row_id).css('display') === 'none'){
$('#top'+row_id).show();
$('#chevron'+row_id).removeClass('glyphicon-chevron-right');
$('#chevron'+row_id).addClass('glyphicon-chevron-down');
} else {
$('#top'+row_id).hide();
$('#chevron'+row_id).removeClass('glyphicon-chevron-down');
$('#chevron'+row_id).addClass('glyphicon-chevron-right');
}
}

function search_data() {
$("#viewer").css("display", "none");
$("#viewer_2").css("display", "none");
$('body').addClass("loading");
$("#result").empty();
$.getJSON($SCRIPT_ROOT + '/search', {
query: $.trim($('input[name="query"]').val())
}).done(function(data) {
search_data_length = Object.keys(data).length;
if(search_data_length == 0){
$("#result").text("No results found");
} else {

$("#result").append("<h2><strong>Data Analysis</strong><h2>");
console.log(Object.keys(data));
console.log(Object.values(data));

for(i=0; i<Object.keys(data).length; i++)
{
$('#result').append("<table> <tr> <h4> <strong>"+ Object.keys(data)[i]+"</strong></h4>"+Object.values(data)[i]+"<br/></tr>");
}

$("#table_results").append("</table>");
$("#viewer").css("display", "contents");
}

$("#viewer_2").css("display", "contents");
}).fail(function () {
$("#result").text("Error when Calling API");
}).always(function () {$('body').removeClass("loading"); });
}

$(function() {
$('#search_box').find('#search_button, #query').bind('keypress', function () {
if (event.which == 13) {
search_data();
return;
}
});
$('#search_box').find('#search_button').bind('click', search_data);
});
</script>
